# this
this는 함수 내에서 함수 호출 맥락(context)를 의미한다. 맥락이라는 것은 상황에 따라서 달라진다는 의미인데 즉 함수를 어떻게 호출하느냐에 따라서 this가 가리키는 대상이 달라진다는 뜻이다. 함수와 객체의 관계가 느슨한 자바스크립트에서 this는 이 둘을 연결시켜주는 실질적인 연결점의 역할을 한다.


{% highlight js %}
var funcThis = null; 
 
function Func(){
    funcThis = this;
}
var o1 = Func(); // this 가 window
if(funcThis === window){
    document.write('window <br />');
}

var o2 = new Func(); // o2
if(funcThis === o2){
    document.write('o2 <br />');
}
{% endhighlight %}
this는 함수를 실행할 때 어떤 생성자를 통해 만들어졌는지를 표시함

## apply
apply 를 통해 함수의 this 를 주입할 수 있음.
{% highlight js %}
var o = {}
var p = {}
function func(){
    switch(this){
        case o:
            document.write('o<br />');
            break;
        case p:
            document.write('p<br />');
            break;
        case window:
            document.write('window<br />');
            break;          
    }
}
func();
func.apply(o);
func.apply(p);
{% endhighlight %}
